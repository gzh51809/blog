### 杨辉三角

- 请用JS实现杨辉三角

```js

var row = 10,
 array = new Array()
 
 for(var i=0;i<row;i++){
     var arraySon = []
     for(var j=0;j<i+1;j++){
         console.log(`i=${i},j=${j}`)
         if(i==0||j==0||i==j){
             arraySon.push(1)
         }else{
             arraySon.push(array[i-1][j-1]+array[i-1][j+1])
         }
     }
     console.log(`arraySon = ${arraySon}`)
     // console.log(arraySon)
     array.push(arraySon)
 }
```
